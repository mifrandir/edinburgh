\begin{frame}
    \frametitle{Definition}
    Let $E\subseteq\mathbb{R}$ be nonempty and let $(f_n:E\to\mathbb{R})$ be a sequence of functions.

    \begin{itemize}
        \item We say $f_n\to f$ pointwise as $n\to\infty$ if \begin{align*}
                  \forall \varepsilon > 0.\:
                  \underline{\mathbf{\forall x\in E.\:\exists N\in\mathbb{N}.}}\:
                  \forall n>N.\:\left\vert f_n(x)-f(x) \right\vert < \varepsilon.
              \end{align*}
        \item We say $f_n\to f$ uniformly as $n\to\infty$ if \begin{align*}
                  \forall \varepsilon >0.\:
                  \underline{\mathbf{\exists N\in\mathbb{N}.\:\forall x\in E.}}\:
                  \forall n > N.\: \left\vert f_n(x)-f(x) \right\vert < \varepsilon.
              \end{align*}
    \end{itemize}
\end{frame}
